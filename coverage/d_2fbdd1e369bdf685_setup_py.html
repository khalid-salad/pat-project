234def check_ieee_macros(config): 

235 priv = [] 

236 pub = [] 

237 

238 macros = [] 

239 

240 def _add_decl(f): 

241 priv.append(fname2def("decl_%s" % f)) 

242 pub.append('NPY_%s' % fname2def("decl_%s" % f))

244 # XXX: hack to circumvent cpp pollution from python: python put its

257 else: 

258 macros.append(f)

260 # func, or both func and macro version. Check for macro only, and define 

261 # replacement ones if not found.

271def check_types(config_cmd, ext, build_dir): 

272 private_defines = [] 

273 public_defines = [] 

274 

275 # Expected size (in number of bytes) for each type. This is an 

276 # optimization: those are only hints, and an exhaustive search for the size 

277 # is done if the hints are wrong. 

278 expected = {'short': [2], 'int': [4], 'long': [8, 4], 

279 'float': [4], 'double': [8], 'long double': [16, 12, 8], 

280 'Py_intptr_t': [8, 4], 'PY_LONG_LONG': [8], 'long long': [8], 

281 'off_t': [8, 4]} 

282 

283 # Check we have the python header (-dev* packages on Linux) 

284 result = config_cmd.check_header('Python.h') 

285 if not result: 

286 python = 'python' 

287 if '__pypy__' in sys.builtin_module_names: 

288 python = 'pypy' 

289 raise SystemError( 

290 "Cannot compile 'Python.h'. Perhaps you need to " 

291 "install {0}-dev|{0}-devel.".format(python)) 

292 res = config_cmd.check_header("endian.h") 

293 if res: 

294 private_defines.append(('HAVE_ENDIAN_H', 1)) 

295 public_defines.append(('NPY_HAVE_ENDIAN_H', 1)) 

296 res = config_cmd.check_header("sys/endian.h") 

297 if res: 

298 private_defines.append(('HAVE_SYS_ENDIAN_H', 1)) 

299 public_defines.append(('NPY_HAVE_SYS_ENDIAN_H', 1)) 

300 

301 # Check basic types sizes 

302 for type in ('short', 'int', 'long'): 

303 res = config_cmd.check_decl("SIZEOF_%s" % sym2def(type), headers=["Python.h"]) 

304 if res: 

305 public_defines.append(('NPY_SIZEOF_%s' % sym2def(type), "SIZEOF_%s" % sym2def(type))) 

306 else: 

307 res = config_cmd.check_type_size(type, expected=expected[type]) 

308 if res >= 0: 

309 public_defines.append(('NPY_SIZEOF_%s' % sym2def(type), '%d' % res)) 

310 else: 

311 raise SystemError("Checking sizeof (%s) failed !" % type) 

312 

313 for type in ('float', 'double', 'long double'): 

314 already_declared = config_cmd.check_decl("SIZEOF_%s" % sym2def(type), 

315 headers=["Python.h"]) 

316 res = config_cmd.check_type_size(type, expected=expected[type]) 

317 if res >= 0: 

318 public_defines.append(('NPY_SIZEOF_%s' % sym2def(type), '%d' % res)) 

319 if not already_declared and not type == 'long double': 

320 private_defines.append(('SIZEOF_%s' % sym2def(type), '%d' % res)) 

321 else: 

322 raise SystemError("Checking sizeof (%s) failed !" % type) 

323 

324 # Compute size of corresponding complex type: used to check that our 

325 # definition is binary compatible with C99 complex type (check done at 

326 # build time in npy_common.h) 

327 complex_def = "struct {%s __x; %s __y;}" % (type, type) 

328 res = config_cmd.check_type_size(complex_def, 

329 expected=[2 * x for x in expected[type]]) 

330 if res >= 0: 

331 public_defines.append(('NPY_SIZEOF_COMPLEX_%s' % sym2def(type), '%d' % res)) 

332 else: 

333 raise SystemError("Checking sizeof (%s) failed !" % complex_def) 

334 

335 for type in ('Py_intptr_t', 'off_t'): 

336 res = config_cmd.check_type_size(type, headers=["Python.h"], 

337 library_dirs=[pythonlib_dir()], 

338 expected=expected[type]) 

339 

340 if res >= 0: 

341 private_defines.append(('SIZEOF_%s' % sym2def(type), '%d' % res)) 

342 public_defines.append(('NPY_SIZEOF_%s' % sym2def(type), '%d' % res)) 

343 else: 

344 raise SystemError("Checking sizeof (%s) failed !" % type) 

345 

346 # We check declaration AND type because that's how distutils does it. 

347 if config_cmd.check_decl('PY_LONG_LONG', headers=['Python.h']): 

348 res = config_cmd.check_type_size('PY_LONG_LONG', headers=['Python.h'], 

349 library_dirs=[pythonlib_dir()], 

350 expected=expected['PY_LONG_LONG']) 

351 if res >= 0: 

352 private_defines.append(('SIZEOF_%s' % sym2def('PY_LONG_LONG'), '%d' % res)) 

353 public_defines.append(('NPY_SIZEOF_%s' % sym2def('PY_LONG_LONG'), '%d' % res)) 

354 else: 

355 raise SystemError("Checking sizeof (%s) failed !" % 'PY_LONG_LONG') 

356 

357 res = config_cmd.check_type_size('long long', 

358 expected=expected['long long']) 

359 if res >= 0: 

360 #private_defines.append(('SIZEOF_%s' % sym2def('long long'), '%d' % res)) 

361 public_defines.append(('NPY_SIZEOF_%s' % sym2def('long long'), '%d' % res)) 

362 else: 

363 raise SystemError("Checking sizeof (%s) failed !" % 'long long') 

364 

365 if not config_cmd.check_decl('CHAR_BIT', headers=['Python.h']): 

366 raise RuntimeError( 

367 "Config wo CHAR_BIT is not supported" 

368 ", please contact the maintainers") 

369 

370 return private_defines, public_defines

487

508 else:

510 with open(target) as target_f:

525 if incl_dir not in config.numpy_include_dirs: 

526 config.numpy_include_dirs.append(incl_dir) 

527
